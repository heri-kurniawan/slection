#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

int main(int argc, char **argv)
{
    system("color F9");
    
    cout<<"Selection Sort \n";
    cout<<"-------------- \n\n";
    
    int banyak_data,key,i,j,tampung;
    
    //Proses Input
    cout<<"Inputkan Banyak Data : ";
    cin>>banyak_data;
    
    int data[banyak_data];
    srand(time(NULL));
    
    for(i=0;i<banyak_data;i++)
    {
        data[i]=rand()%100;
    }
    
    //Menampilkan Data Sebelm Diurut
    cout<<"Data Sebelum Diurutkan \n";
    int sbl;
    for(sbl=0;sbl<banyak_data;sbl++)
    {
        cout<<data[sbl]<<" ";
    }
    cout<<endl<<endl;

    for(i=0;i<banyak_data;i++)
    {
        key=i;
        for(j=i+1;j<banyak_data;j++)
        {
            if(data[key]>data[j])
            {
                key=j;
            }
        }
        tampung=data[key];
        data[key]=data[i];
        data[i]=tampung;
    }
    
    //Menampilkan Data Setelah Diurut
    cout<<"Hasil Data Sesudah Diurutkan \n";
    int ssd;
    for(ssd=0;ssd<banyak_data;ssd++)
    {
        cout<<data[ssd]<<" ";
    }

    cout<<endl<<endl;
	return 0;
}
